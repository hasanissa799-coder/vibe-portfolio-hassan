    <script>
        const titles = [
            "مهندس ومطور حلول متكاملة", 
            "خبير الأتمتة والذكاء الاصطناعي", 
            "مطور No-Code / Low-Code", 
            "مبتكر بتقنية Vibe Coding"
        ];
        
        let titleIndex = 0;
        let charIndex = 0;
        let isDeleting = false;
        const typeWriterElement = document.getElementById("typewriter");

        function type() {
            const currentTitle = titles[titleIndex];
            
            if (isDeleting) {
                typeWriterElement.textContent = currentTitle.substring(0, charIndex - 1);
                charIndex--;
            } else {
                typeWriterElement.textContent = currentTitle.substring(0, charIndex + 1);
                charIndex++;
            }

            let typeSpeed = 80; // سرعة الكتابة

            if (isDeleting) {
                typeSpeed /= 2; // سرعة الحذف
            }

            if (!isDeleting && charIndex === currentTitle.length) {
                typeSpeed = 2000; // فترة الانتظار بعد اكتمال الكلمة
                isDeleting = true;
            } else if (isDeleting && charIndex === 0) {
                isDeleting = false;
                titleIndex = (titleIndex + 1) % titles.length;
                typeSpeed = 500; // فترة الانتظار قبل الكلمة الجديدة
            }

            setTimeout(type, typeSpeed);
        }

        // بدء التأثير بعد تحميل الصفحة
        document.addEventListener("DOMContentLoaded", () => {
            setTimeout(type, 1000);
        });
    </script>
